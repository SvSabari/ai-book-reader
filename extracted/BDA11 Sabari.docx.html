Real-Time News Trend and Sentiment Analyzer using Python

PROGRAM:

import requests from bs4 import BeautifulSoup from collections import Counter from textblob import TextBlob import matplotlib.pyplot as plt import re print("Fetching Latest News...\n")

# Fetch news url = "https://news.ycombinator.com/" response = requests.get(url) soup = BeautifulSoup(response.text, "html.parser")

# Extract headlines headlines = [] for tag in soup.select(".titleline > a"):

headlines.append(tag.text) print("Latest Headlines:\n") for h in headlines[:10]:

print("-", h) # Extract words all_words = [] for headline in headlines: words = re.findall(r'\b[A-Za-z]+\b', headline.lower() all_words.extend(words) stopwords = ["the","is","a","an","of","to","and","in","for","on","with","at","by"] filtered_words = [w for w in all_words if w not in stopwords]

# Word frequency word_count = Counter(filtered_words) top_words = word_count.most_common(10) print("\nTop Trending Words:\n") for word, count in top_words: print(word, ":", count) # Sentiment analysis sentiments = {"Positive":0, "Negative":0, "Neutral":0} print("\nSentiment Analysis:\n") for headline in headlines[:10]: analysis = TextBlob(headline) if analysis.sentiment.polarity > 0: sentiment = "Positive" sentiments["Positive"] += 1 elif analysis.sentiment.polarity < 0: sentiment = "Negative" sentiments["Negative"] += 1 else:

sentiment = "Neutral" sentiments["Neutral"] += 1 print(headline, "-->", sentiment) words = [w[0] for w in top_words] counts = [w[1] for w in top_words] plt.bar(words, counts) plt.title("Trending Topics in News") plt.xlabel("Words") plt.ylabel("Frequency") plt.xticks(rotation=45) plt.show() labels = sentiments.keys()

sizes = sentiments.values() plt.pie(sizes, labels=labels, autopct='%1.1f%%') plt.title("News Sentiment Distribution") plt.show()
